**1.6: `lib/brand-voice/build-system-blocks.ts`** — KEY FILE. Implement:
```typescript
import type { SystemBlock } from '@anthropic-ai/sdk/resources/messages'

export function buildSystemBlocks(
  orgId: string,
  agentInstructions: string,
  brandVoicePrompt: string | null,
): SystemBlock[] {
  const blocks: SystemBlock[] = [
    // Block 0: org_id — DISTINCT first block forces tenant-scoped cache key (VOICE-04, Pitfall 4)
    { type: 'text', text: `TENANT_CONTEXT: org_id=${orgId}` },
    // Block 1: agent-specific instructions (no cache — varies per agent type)
    { type: 'text', text: agentInstructions },
  ]
  if (brandVoicePrompt) {
    // Block 2: brand voice — cached, padded to >=4096 token floor
    blocks.push({
      type: 'text',
      text: brandVoicePrompt, // assumed already padded by save route
      cache_control: { type: 'ephemeral' },
    })
  }
  return blocks
}
```

CRITICAL: do NOT add cache_control on block 0 or 1. Anthropic caches up to and including the LAST block that has cache_control set. Putting cache_control only on block 2 means blocks 0+1+2 are all part of the cached prefix. The org_id at block 0 forces a unique cache key per tenant (VOICE-04), which is the entire point.

Also export the inverse helper for tests:
```typescript
export function extractOrgIdFromBlocks(blocks: SystemBlock[]): string | null {
  const first = blocks[0]
  if (!first || first.type !== 'text') return null
  const m = first.text.match(/^TENANT_CONTEXT: org_id=(.+)$/)
  return m?.[1] ?? null
}
```

import { describe, it, expect } from 'vitest'
import { getAdminClient } from '@/lib/supabase/admin'
import Anthropic from '@anthropic-ai/sdk'
import { buildSystemBlocks } from '@/lib/brand-voice'
import { padToCacheFloor } from '@/lib/brand-voice/pad-to-cache'

const ORG_A = process.env.TEST_CACHE_ORG_A
const ORG_B = process.env.TEST_CACHE_ORG_B

describe.skipIf(!ORG_A || !ORG_B)('VOICE-05 golden two-tenant cache isolation', () => {
  const anthropic = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY! })

  // Setup: assert both orgs have DIFFERENT brand voices in client_profiles
  // (test runner must seed these; document in test header)

  // Voice fixtures include UNAMBIGUOUS signature phrases that must appear in any output —
  // this lets the test assert markers (defeats short-output flakiness from a 1-sentence prompt).
  const VOICE_A = padToCacheFloor('You are bold and direct. Use exclamation marks. Every reply MUST end with the literal phrase "Hustle on!" — no exceptions, no rephrasing.')
  const VOICE_B = padToCacheFloor('You are warm and gentle. Use ellipses... Every reply MUST end with the literal phrase "with care." — no exceptions, no rephrasing.')
  const PROMPT = 'Write a 1-sentence welcome message for a new client.'

  it('two orgs with different voices produce different outputs', async () => {
    const blocksA = buildSystemBlocks(ORG_A!, 'Welcome agent', VOICE_A)
    const blocksB = buildSystemBlocks(ORG_B!, 'Welcome agent', VOICE_B)
    
    const [respA, respB] = await Promise.all([
      anthropic.messages.create({
        model: 'claude-haiku-4-5-20251001',
        max_tokens: 200,
        system: blocksA,
        messages: [{ role: 'user', content: PROMPT }],
      }),
      anthropic.messages.create({
        model: 'claude-haiku-4-5-20251001',
        max_tokens: 200,
        system: blocksB,
        messages: [{ role: 'user', content: PROMPT }],
      }),
    ])

    const textA = respA.content[0].type === 'text' ? respA.content[0].text : ''
    const textB = respB.content[0].type === 'text' ? respB.content[0].text : ''

    // Outputs must differ — different voices, different blocks 0
    expect(textA).not.toBe(textB)
    // Marker assertions: signature phrases prove each output reflects its voice.
    // (Anti-flakiness: a 1-sentence prompt could in theory produce identical-by-chance
    // outputs. Markers are the strong assertion; .not.toBe is the weak one.)
    expect(textA.toLowerCase()).toContain('hustle on')
    expect(textB.toLowerCase()).toContain('with care')
  }, 30_000)

  it('second call same org gets cache hit (cache_read_input_tokens > 0)', async () => {
    const blocks = buildSystemBlocks(ORG_A!, 'Welcome agent', VOICE_A)
    
    // First call — establishes cache (cache_creation_input_tokens > 0)
    const first = await anthropic.messages.create({
      model: 'claude-haiku-4-5-20251001',
      max_tokens: 200,
      system: blocks,
      messages: [{ role: 'user', content: PROMPT + ' (call 1)' }],
    })
    
    // Second call — should HIT cache
    const second = await anthropic.messages.create({
      model: 'claude-haiku-4-5-20251001',
      max_tokens: 200,
      system: blocks,
      messages: [{ role: 'user', content: PROMPT + ' (call 2)' }],
    })

    expect(second.usage.cache_read_input_tokens ?? 0).toBeGreaterThan(0)
  }, 30_000)
})

**3.1: `app/api/brand-voice/scrape/route.ts` — POST**

```typescript
import { NextRequest } from 'next/server'
import { getUserOrg } from '@/lib/auth/get-user-org'
import { scrapeWebsiteContext } from '@/lib/brand-voice/scraper'
import { z } from 'zod'

const ScrapeInputSchema = z.object({
  url: z.string().url(),
})

export async function POST(req: NextRequest) {
  const { data: userOrg, error: authError } = await getUserOrg()
  if (authError || !userOrg) return Response.json({ error: authError?.message ?? 'unauthenticated' }, { status: 401 })

  const body = await req.json().catch(() => null)
  const parsed = ScrapeInputSchema.safeParse(body)
  if (!parsed.success) return Response.json({ error: 'invalid_input', issues: parsed.error.issues }, { status: 400 })

  // Block private IPs to prevent SSRF
  const u = new URL(parsed.data.url)
  if (['localhost','127.0.0.1','0.0.0.0'].includes(u.hostname) || u.hostname.startsWith('10.') || u.hostname.startsWith('192.168.') || u.hostname.startsWith('172.')) {
    return Response.json({ error: 'private_url_not_allowed' }, { status: 400 })
  }

  try {
    const ctx = await scrapeWebsiteContext(parsed.data.url)
    return Response.json(ctx)
  } catch (err) {
    return Response.json({ error: 'scrape_failed', detail: (err as Error).message }, { status: 502 })
  }
}
```

SSRF guard is mandatory — without it a tenant can hit internal infra via the scraper.

**3.2: `app/api/brand-voice/save/route.ts` — POST**

```typescript
import { NextRequest } from 'next/server'
import { getUserOrg } from '@/lib/auth/get-user-org'
import { getAdminClient } from '@/lib/supabase/admin'
import { BrandVoiceInputSchema } from '@/lib/brand-voice/wizard-questions'
import { assembleBrandVoicePrompt } from '@/lib/brand-voice/assemble-prompt'
import { scrubPII } from '@/lib/brand-voice/pii-scrubber'
import { padToCacheFloor } from '@/lib/brand-voice/pad-to-cache'

export async function POST(req: NextRequest) {
  const { data: userOrg, error: authError } = await getUserOrg()
  if (authError || !userOrg) return Response.json({ error: 'unauthenticated' }, { status: 401 })

  const body = await req.json().catch(() => null)
  const parsed = BrandVoiceInputSchema.safeParse(body)
  if (!parsed.success) return Response.json({ error: 'invalid_input', issues: parsed.error.issues }, { status: 400 })

  // Sequencing: assemble → scrub → pad. Order matters: scrub the assembled doc to catch
  // PII slipped into questions; then pad after, since padding is stable text only.
  const assembled = assembleBrandVoicePrompt(parsed.data, userOrg.organizationName ?? 'Business')
  const scrubbed = scrubPII(assembled)
  const padded = padToCacheFloor(scrubbed)

  const supa = getAdminClient()
  const { error: upsertError } = await supa
    .from('client_profiles')
    .upsert(
      {
        organization_id: userOrg.organizationId,
        brand_voice_prompt: padded,
        example_phrases: parsed.data.example_phrases,
        forbidden_topics: parsed.data.forbidden_topics,
        brand_voice_updated_at: new Date().toISOString(),
      },
      { onConflict: 'organization_id' },
    )
  if (upsertError) return Response.json({ error: 'save_failed', detail: upsertError.message }, { status: 500 })

  return Response.json({ ok: true, updated_at: new Date().toISOString() })
}
```

VOICE-08 (re-run wizard) is naturally supported by the upsert: a second save overwrites the prior brand_voice_prompt and bumps brand_voice_updated_at. BaseAgent's per-instance brand voice cache is bounded by request lifetime, so the next agent invocation in a new request loads the fresh voice.

export async function GET() {
  const { data: userOrg, error: authError } = await getUserOrg()
  if (authError || !userOrg) return Response.json({ error: 'unauthenticated' }, { status: 401 })

  const supa = getAdminClient()
  const { data, error } = await supa
    .from('client_profiles')
    .select('brand_voice_prompt, example_phrases, forbidden_topics, brand_voice_updated_at')

  return Response.json(data)
}
