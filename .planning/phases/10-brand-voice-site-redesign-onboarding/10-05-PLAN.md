```typescript
import { getAdminClient } from '@/lib/supabase/admin'

export interface CostRow {
  orgId: string
  orgName: string
  subdomain: string | null
  costMTDZarCents: number       // sum of daily_cost_rollup.total_cost_zar_cents this month
  mrrZarCents: number           // subscription_composition.monthly_total_zar_cents (current)
  marginPct: number             // (mrr - cost_mtd) / mrr * 100; null-safe → 0 if mrr=0
  isOverFortyPctMrrFlag: boolean // EXACTLY: cost_mtd_cents * 12 > mrr_annual_cents * 0.40
                                  // i.e., monthly_cost > monthly_mrr * 0.40
  last30DaysCostTrend: { date: string; costZarCents: number }[] // 30 entries, oldest first
}

/**
 * 40% MRR flag formula (USAGE-11 spec):
 *   monthly_cost_zar_cents > monthly_mrr_zar_cents * 0.40
 *
 * The annualized form (cost_mtd_cents * 12 > annual_mrr_cents * 0.40) is mathematically
 * equivalent for full months and gives a more conservative early warning mid-month.
 *
 * For v3.0: use the simple monthly form. cost_mtd is partial-month so the flag fires LATER
 * in the month than ideal — but it never fires falsely (false negative bias is acceptable).
 */
export function isOverFortyPctMrrFlag(costMTDZarCents: number, mrrZarCents: number): boolean {
  if (mrrZarCents <= 0) return costMTDZarCents > 0  // any cost on a free org is a flag
  return costMTDZarCents > mrrZarCents * 0.40
}

export function computeMarginPct(costZarCents: number, mrrZarCents: number): number {
  if (mrrZarCents <= 0) return 0
  return ((mrrZarCents - costZarCents) / mrrZarCents) * 100
}

export async function getCostMonitoringRows(): Promise<CostRow[]> {
  const supa = getAdminClient()

  // 1. Active orgs (excluding archived from 10-01 migration 33)
  const { data: orgs, error: orgErr } = await supa
    .from('organizations')
    .select('id, name, subdomain')
    .is('archived_at', null)
  if (orgErr) throw new Error(`orgs query: ${orgErr.message}`)
  if (!orgs) return []

  // 2. Current composition per org (subscription_composition WHERE effective_to IS NULL)
  const { data: comps } = await supa
    .from('subscription_composition')
    .select('organization_id, monthly_total_zar_cents')
    .is('effective_to', null)
  const mrrByOrg = new Map<string, number>()
  for (const c of comps ?? []) mrrByOrg.set(c.organization_id, c.monthly_total_zar_cents)

  // 3. Cost MTD per org — daily_cost_rollup SUM since month start
  const monthStart = new Date()
  monthStart.setDate(1)
  monthStart.setHours(0, 0, 0, 0)
  const { data: mtdRows } = await supa
    .from('daily_cost_rollup')
    .select('organization_id, total_cost_zar_cents')
    .gte('rollup_date', monthStart.toISOString().slice(0, 10))
  const costByOrg = new Map<string, number>()
  for (const r of mtdRows ?? []) {
    costByOrg.set(r.organization_id, (costByOrg.get(r.organization_id) ?? 0) + r.total_cost_zar_cents)
  }

  // 4. 30-day trend per org — daily_cost_rollup last 30 days
  const trendStart = new Date()
  trendStart.setDate(trendStart.getDate() - 30)
  const { data: trendRows } = await supa
    .from('daily_cost_rollup')
    .select('organization_id, rollup_date, total_cost_zar_cents')
    .gte('rollup_date', trendStart.toISOString().slice(0, 10))
    .order('rollup_date', { ascending: true })
  const trendByOrg = new Map<string, { date: string; costZarCents: number }[]>()
  for (const r of trendRows ?? []) {
    const arr = trendByOrg.get(r.organization_id) ?? []
    arr.push({ date: r.rollup_date, costZarCents: r.total_cost_zar_cents })
    trendByOrg.set(r.organization_id, arr)
  }

  // 5. Assemble
  return orgs.map((o) => {
    const cost = costByOrg.get(o.id) ?? 0
    const mrr = mrrByOrg.get(o.id) ?? 0
    return {
      orgId: o.id,
      orgName: o.name,
      subdomain: o.subdomain,
      costMTDZarCents: cost,
      mrrZarCents: mrr,
      marginPct: computeMarginPct(cost, mrr),
      isOverFortyPctMrrFlag: isOverFortyPctMrrFlag(cost, mrr),
      last30DaysCostTrend: trendByOrg.get(o.id) ?? [],
    }
  }).sort((a, b) => a.marginPct - b.marginPct) // worst-margin first
}
```

**1.2: `app/api/admin/cost-monitoring/route.ts`**

```typescript
import { getUserOrg } from '@/lib/auth/get-user-org'
import { getCostMonitoringRows } from '@/lib/admin/cost-monitoring'

export async function GET() {
  const { data: userOrg, error: authError } = await getUserOrg()
  if (authError || !userOrg) return Response.json({ error: 'unauthenticated' }, { status: 401 })

  // user_role enum is {admin, manager, user, client} per Phase 09-01 decision.
  // Platform admin = role 'admin' AND org is the platform_admin org (DragoonB Business Automation).
  // For v3.0 simplicity: any 'admin' role in any org sees this page (single-operator deployment).
  // If multi-operator: tighten to org subdomain check or a platform_admin_user table later.
  if (userOrg.role !== 'admin' && userOrg.role !== 'platform_admin') {
    return Response.json({ error: 'forbidden' }, { status: 403 })
  }

  try {
    const rows = await getCostMonitoringRows()
    return Response.json({ rows, generatedAt: new Date().toISOString() })
  } catch (err) {
    return Response.json({ error: 'fetch_failed', detail: (err as Error).message }, { status: 500 })
  }
}
```

```typescript
import { isOverFortyPctMrrFlag, computeMarginPct } from '@/lib/admin/cost-monitoring'

describe('USAGE-11 40% MRR flag formula', () => {
  // 40% MRR flag: monthly_cost > monthly_mrr * 0.40
  it('NOT flagged: cost = 30% of MRR', () => {
    expect(isOverFortyPctMrrFlag(30000, 100000)).toBe(false)
  })
  it('NOT flagged: cost EXACTLY 40% of MRR (strict greater-than)', () => {
    expect(isOverFortyPctMrrFlag(40000, 100000)).toBe(false)
  })
  it('FLAGGED: cost = 41% of MRR', () => {
    expect(isOverFortyPctMrrFlag(41000, 100000)).toBe(true)
  })
  it('FLAGGED: cost > 0 with MRR = 0 (free org with cost is always a flag)', () => {
    expect(isOverFortyPctMrrFlag(100, 0)).toBe(true)
  })
  it('NOT flagged: cost = 0 with MRR = 0', () => {
    expect(isOverFortyPctMrrFlag(0, 0)).toBe(false)
  })
})

describe('margin calculation', () => {
  it('returns positive margin when cost < MRR', () => {
    expect(computeMarginPct(20000, 100000)).toBeCloseTo(80)
  })
  it('returns 0 when MRR is 0 (avoids divide-by-zero)', () => {
    expect(computeMarginPct(100, 0)).toBe(0)
  })
  it('returns negative margin when cost exceeds MRR', () => {
    expect(computeMarginPct(150000, 100000)).toBeCloseTo(-50)
  })
})
```

```typescript
'use client'
import { useMemo, useState } from 'react'
import { useReactTable, getCoreRowModel, flexRender, ColumnDef } from '@tanstack/react-table'
import type { CostRow } from '@/lib/admin/cost-monitoring'
import { CostTrendChart } from './cost-trend-chart'

function formatZAR(cents: number): string {
  return `R${(cents / 100).toLocaleString('en-ZA', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}`
}

export function CostTable({ rows }: { rows: CostRow[] }) {
  const [expandedOrgId, setExpandedOrgId] = useState<string | null>(null)

  const columns: ColumnDef<CostRow>[] = useMemo(() => [
    {
      header: 'Org',
      accessorKey: 'orgName',
      cell: ({ row }) => (
        <div>
          <div className="font-medium">{row.original.orgName}</div>
          <div className="text-xs text-gray-500">{row.original.subdomain}</div>
        </div>
      ),
    },
    { header: 'MRR', accessorFn: (r) => formatZAR(r.mrrZarCents) },
    { header: 'Cost MTD', accessorFn: (r) => formatZAR(r.costMTDZarCents) },
    {
      header: 'Margin %',
      accessorFn: (r) => `${r.marginPct.toFixed(1)}%`,
      cell: ({ row }) => {
        const m = row.original.marginPct
        const color = m < 0 ? 'text-red-600' : m < 60 ? 'text-amber-600' : 'text-green-700'
        return <span className={color}>{m.toFixed(1)}%</span>
      },
    },
    {
      header: 'Flag',
      cell: ({ row }) => row.original.isOverFortyPctMrrFlag
        ? <span className="px-2 py-1 text-xs bg-[#6B1420] text-white rounded">&gt;40% MRR</span>
        : <span className="text-gray-400">—</span>,
    },
    {
      header: 'Trend',
      cell: ({ row }) => (
        <button
          onClick={() => setExpandedOrgId(expandedOrgId === row.original.orgId ? null : row.original.orgId)}
          className="text-sm text-[#6B1420] underline"
        >{expandedOrgId === row.original.orgId ? 'Hide' : 'Show 30d'}</button>
      ),
    },
  ], [expandedOrgId])

  const table = useReactTable({ data: rows, columns, getCoreRowModel: getCoreRowModel() })

  return (
    <div className="rounded-lg border border-gray-200 bg-white">
      <table className="w-full">
        <thead className="bg-gray-50 text-left text-sm">
          {table.getHeaderGroups().map((hg) => (
            <tr key={hg.id}>
              {hg.headers.map((h) => (
                <th key={h.id} className="p-3">{flexRender(h.column.columnDef.header, h.getContext())}</th>
              ))}
            </tr>
          ))}
        </thead>
        <tbody>
          {table.getRowModel().rows.map((r) => (
            <>
              <tr key={r.id} className="border-t hover:bg-gray-50">
                {r.getVisibleCells().map((c) => (
                  <td key={c.id} className="p-3 text-sm">{flexRender(c.column.columnDef.cell, c.getContext())}</td>
                ))}
              </tr>
              {expandedOrgId === r.original.orgId && (
                <tr key={r.id + '-trend'}>
                  <td colSpan={6} className="p-4 bg-gray-50">
                    <CostTrendChart trend={r.original.last30DaysCostTrend} />
                  </td>
                </tr>
              )}
            </>
          ))}
        </tbody>
      </table>
      {rows.length === 0 && <p className="p-6 text-center text-gray-500">No active orgs with cost data yet.</p>}
    </div>
  )
}
```

```typescript
'use client'
import { LineChart, Line, XAxis, YAxis, Tooltip, ResponsiveContainer, CartesianGrid } from 'recharts'

export function CostTrendChart({ trend }: { trend: { date: string; costZarCents: number }[] }) {
  const data = trend.map((t) => ({ date: t.date.slice(5), costRand: t.costZarCents / 100 }))
  return (
    <div className="h-48 w-full">
      <ResponsiveContainer width="100%" height="100%">
        <LineChart data={data}>
          <CartesianGrid strokeDasharray="3 3" />
          <XAxis dataKey="date" tick={{ fontSize: 11 }} />
          <YAxis tick={{ fontSize: 11 }} tickFormatter={(v) => `R${v.toFixed(0)}`} />
          <Tooltip formatter={(v: number) => `R${v.toFixed(2)}`} />
          <Line type="monotone" dataKey="costRand" stroke="#6B1420" strokeWidth={2} dot={false} />
        </LineChart>
      </ResponsiveContainer>
    </div>
  )
}
```
