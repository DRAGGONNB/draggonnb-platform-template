```typescript
/**
 * VAT-inclusive pricing for South African ZAR amounts.
 * 15% VAT is the standard SA rate. Always integer cent math — never float.
 * BILL-09 spec: "displays VAT-inclusive ZAR amount with a clear 'incl. 15% VAT' line".
 */
const VAT_RATE_BASIS_POINTS = 1500 // 15.00%

export function vatInclusivePrice(exclusiveZarCents: number): number {
  // Math.round(cents * 1.15) — equivalent to (cents * 11500 + 5000) / 10000 for pure integer math
  return Math.round(exclusiveZarCents * (1 + VAT_RATE_BASIS_POINTS / 10000))
}

export function vatPortion(exclusiveZarCents: number): number {
  return vatInclusivePrice(exclusiveZarCents) - exclusiveZarCents
}

export const VAT_LABEL = 'incl. 15% VAT'
```

**1.2: `lib/billing/format-zar.ts`** (if not already exists; check first)

```typescript
export function formatZAR(cents: number): string {
  const rands = cents / 100
  return `R${rands.toLocaleString('en-ZA', { minimumFractionDigits: 0, maximumFractionDigits: 0 })}`
}
export function formatZARDecimal(cents: number): string {
  return `R${(cents / 100).toLocaleString('en-ZA', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}`
}
```

```typescript
'use client'
import { useState, useMemo } from 'react'
import { vatInclusivePrice, VAT_LABEL } from '@/lib/billing/vat'
import { formatZARDecimal } from '@/lib/billing/format-zar'

interface Plan { id: string; name: string; priceZarCents: number; description: string }
interface Addon { id: string; display_name: string; price_zar_cents: number; kind: string; billing_cycle: string }

export function ModulePicker({ plans, addons }: { plans: Plan[]; addons: Addon[] }) {
  const [selectedPlanId, setSelectedPlanId] = useState<string>('core')
  const [selectedAddonIds, setSelectedAddonIds] = useState<Set<string>>(new Set())

  const monthlyAddons = addons.filter((a) => a.kind === 'module' && a.billing_cycle === 'monthly')
  const setupFee = addons.find((a) => a.id === 'setup_fee')

  const totals = useMemo(() => {
    const plan = plans.find((p) => p.id === selectedPlanId)
    const planCents = plan?.priceZarCents ?? 0
    const addonCents = Array.from(selectedAddonIds).reduce((sum, id) => {
      const a = addons.find((x) => x.id === id)
      return sum + (a?.price_zar_cents ?? 0)
    }, 0)
    const subtotalEx = planCents + addonCents
    const subtotalInc = vatInclusivePrice(subtotalEx)
    const setupEx = setupFee?.price_zar_cents ?? 0
    const setupInc = vatInclusivePrice(setupEx)
    return { subtotalEx, subtotalInc, setupEx, setupInc }
  }, [plans, addons, selectedPlanId, selectedAddonIds, setupFee])

  function toggleAddon(id: string) {
    setSelectedAddonIds((prev) => {
      const next = new Set(prev)
      if (next.has(id)) next.delete(id)
      else next.add(id)
      return next
    })
  }

  return (
    <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
      <div className="lg:col-span-2 space-y-6">
        {/* Plan selector */}
        <fieldset className="space-y-3">
          <legend className="font-semibold text-lg">1. Pick your plan</legend>
          {plans.map((p) => (
            <label key={p.id} className={`block p-4 border-2 rounded-lg cursor-pointer ${selectedPlanId === p.id ? 'border-[#6B1420] bg-[#fef5f6]' : 'border-gray-200'}`}>
              <input type="radio" name="plan" checked={selectedPlanId === p.id} onChange={() => setSelectedPlanId(p.id)} className="sr-only" />
              <div className="flex justify-between items-start">
                <div>
                  <div className="font-semibold">{p.name}</div>
                  <div className="text-sm text-gray-600">{p.description}</div>
                </div>
                <div className="font-mono text-lg">{formatZARDecimal(p.priceZarCents)}/mo</div>
              </div>
            </label>
          ))}
        </fieldset>

        {/* Addons */}
        <fieldset className="space-y-3">
          <legend className="font-semibold text-lg">2. Add modules (optional)</legend>
          {monthlyAddons.map((a) => (
            <label key={a.id} className="block p-4 border border-gray-200 rounded-lg cursor-pointer hover:border-[#6B1420]">
              <div className="flex items-center justify-between">
                <div className="flex items-center gap-3">
                  <input type="checkbox" checked={selectedAddonIds.has(a.id)} onChange={() => toggleAddon(a.id)} />
                  <div>{a.display_name}</div>
                </div>
                <div className="font-mono">{formatZARDecimal(a.price_zar_cents)}/mo</div>
              </div>
            </label>
          ))}
        </fieldset>
      </div>

      {/* Live total sidebar */}
      <aside className="bg-[#363940] text-white p-6 rounded-lg h-fit sticky top-6">
        <h3 className="font-semibold mb-4">Your total</h3>
        <dl className="space-y-2 text-sm">
          <div className="flex justify-between"><dt>Monthly (ex VAT)</dt><dd>{formatZARDecimal(totals.subtotalEx)}</dd></div>
          <div className="flex justify-between text-base font-semibold pt-2 border-t border-gray-600">
            <dt>Monthly</dt>
            <dd>{formatZARDecimal(totals.subtotalInc)}</dd>
          </div>
          <div className="flex justify-between text-xs text-gray-300">
            <dt>{VAT_LABEL}</dt>
            <dd></dd>
          </div>
          {totals.setupEx > 0 && (
            <div className="pt-3 border-t border-gray-600 mt-3 text-xs">
              <div className="flex justify-between"><dt>Once-off setup</dt><dd>{formatZARDecimal(totals.setupInc)}</dd></div>
              <div className="text-gray-400">{VAT_LABEL} — billed after first month</div>
            </div>
          )}
        </dl>
        <div className="mt-6 text-xs text-gray-300">
          Replaces ~R4,500/mo of manual work. Live in 3 business days. Cancel anytime.
        </div>
        <a href="/signup" className="mt-6 block w-full py-3 bg-[#6B1420] text-center font-semibold rounded hover:bg-[#8a1a29]">Get started</a>
      </aside>
    </div>
  )
}
```

```typescript
// (See 3.5: a single client component that hosts the 3 steps via useState; alternatively split as below.)

export const dynamic = 'force-dynamic'

export default async function BrandVoicePage() {
  const { data: userOrg, error } = await getUserOrg()
  if (error || !userOrg) redirect('/login')

  // Pre-fetch existing voice directly via the admin client (server-side).
  // Calling the API with an empty cookie would 401 the request and silently break
  // VOICE-08 re-run mode (the wizard would never show "Last updated" header).
  // The page is already auth-gated above via getUserOrg() so a direct admin-client
  // read is safe and avoids the round-trip + auth-forwarding tax.
  const { createAdminClient } = await import('@/lib/supabase/admin')
```

```typescript
'use client'
import { useState } from 'react'
import { WizardStepQuestions } from './wizard-step-questions'
import type { ScrapedBrandContext } from '@/lib/brand-voice/scraper'

export function BrandVoiceWizard({ initialExistingVoice }: { initialExistingVoice: any }) {
  const [step, setStep] = useState<'url'|'questions'|'review'|'done'>('url')
  const [scraped, setScraped] = useState<ScrapedBrandContext | null>(null)
  const [url, setUrl] = useState('')
  const [scraping, setScraping] = useState(false)
  const [error, setError] = useState<string | null>(null)
  const [answers, setAnswers] = useState<any>(null)

  async function handleScrape() {
    setScraping(true); setError(null)
    try {
      const res = await fetch('/api/brand-voice/scrape', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ url }),
      })
      if (!res.ok) {
        const j = await res.json().catch(() => ({}))
        throw new Error(j.error ?? `HTTP ${res.status}`)
      }
      const data = await res.json()
      setScraped(data)
      setStep('questions')
    } catch (e) {
      setError((e as Error).message)
    } finally {
      setScraping(false)
    }
  }

  async function handleSave() {
    const res = await fetch('/api/brand-voice/save', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ scrapedContext: scraped, ...answers }),
    })
    if (!res.ok) {
      const j = await res.json().catch(() => ({}))
      setError(j.error ?? `HTTP ${res.status}`)
      return
    }
    setStep('done')
  }

  if (step === 'url') {
    return (
      <div className="mt-8 space-y-4">
        <label className="block">
          <span className="font-semibold">Step 1 of 3 — Your website</span>
          <input
            type="url"
            value={url}
            onChange={(e) => setUrl(e.target.value)}
            placeholder="https://yourbusiness.co.za"
            className="mt-2 w-full p-3 border border-gray-300 rounded"
          />
        </label>
        {error && <p className="text-red-600 text-sm">{error}</p>}
        <div className="flex gap-3">
          <button onClick={handleScrape} disabled={!url || scraping} className="px-4 py-2 bg-[#6B1420] text-white rounded disabled:opacity-50">
            {scraping ? 'Scanning…' : 'Continue'}
          </button>
          <button onClick={() => { setScraped(null); setStep('questions') }} className="px-4 py-2 text-gray-600 hover:text-gray-900">
            Skip — fill in manually
          </button>
        </div>
      </div>
    )
  }

  if (step === 'questions') {
    return <WizardStepQuestions scraped={scraped} onSubmit={(a) => { setAnswers(a); setStep('review') }} />
  }

  if (step === 'review') {
    return (
      <div className="mt-8">
        <h2 className="font-semibold">Step 3 of 3 — Review and save</h2>
        <pre className="mt-3 p-4 bg-gray-50 rounded text-xs overflow-auto max-h-96">{JSON.stringify(answers, null, 2)}</pre>
        {error && <p className="text-red-600 text-sm mt-2">{error}</p>}
        <div className="flex gap-3 mt-4">
          <button onClick={() => setStep('questions')} className="px-4 py-2 text-gray-600">Back</button>
          <button onClick={handleSave} className="px-4 py-2 bg-[#6B1420] text-white rounded">Save brand voice</button>
        </div>
      </div>
    )
  }

  if (step === 'done') {
    return (
      <div className="mt-8 p-6 bg-green-50 border border-green-200 rounded">
        <h2 className="font-semibold text-green-900">Brand voice saved</h2>
        <p className="text-sm mt-2">Every AI agent will now use this voice on the next request.</p>
      </div>
    )
  }
  return null
}
```

```typescript
'use client'
import { useEffect, useState } from 'react'
import Link from 'next/link'

interface Progress {
  day0_completed_at: string | null
  brand_voice_completed_at: string | null
  kickoff_call_scheduled_at: string | null
  steps_completed: string[]
}

export function OnboardingChecklist() {
  const [progress, setProgress] = useState<Progress | null>(null)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    fetch('/api/ops/onboarding-progress')
      .then((r) => r.ok ? r.json() : null)
      .then(setProgress)
      .finally(() => setLoading(false))
  }, [])

  if (loading) return <div className="p-4 text-sm text-gray-500">Loading...</div>
  if (!progress || !progress.day0_completed_at) return null // hide for non-onboarding orgs

  // Hide once all 4 steps complete
  const allDone = progress.brand_voice_completed_at && progress.kickoff_call_scheduled_at && progress.steps_completed?.includes('first_action')
  if (allDone) return null

  const items = [
    { label: 'Account live', done: !!progress.day0_completed_at, href: null },
    { label: 'Brand voice captured', done: !!progress.brand_voice_completed_at, href: '/settings/brand-voice' },
    { label: 'Kickoff call booked', done: !!progress.kickoff_call_scheduled_at, href: process.env.NEXT_PUBLIC_KICKOFF_CALL_URL ?? '#' },
    { label: 'First action complete', done: progress.steps_completed?.includes('first_action') ?? false, href: '/dashboard' },
  ]

  return (
    <section className="rounded-lg border border-[#6B1420] bg-[#fef5f6] p-6 mb-6">
      <h3 className="font-semibold text-[#363940]">Get to live in 3 business days</h3>
      <ul className="mt-3 space-y-2 text-sm">
        {items.map((item, i) => (
          <li key={i} className="flex items-center gap-3">
            <span className={`w-5 h-5 rounded-full flex items-center justify-center text-xs ${item.done ? 'bg-green-600 text-white' : 'bg-gray-200'}`}>
              {item.done ? '✓' : i + 1}
            </span>
            {item.href && !item.done
              ? <Link href={item.href} className="underline">{item.label}</Link>
              : <span className={item.done ? 'text-gray-500 line-through' : ''}>{item.label}</span>}
          </li>
        ))}
      </ul>
    </section>
  )
}
```
