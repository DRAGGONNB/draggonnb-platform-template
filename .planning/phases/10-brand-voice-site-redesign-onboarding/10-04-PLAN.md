**1.1: `lib/provisioning/business-days.ts` (ONBOARD-09)**

```typescript
import { formatInTimeZone, toZonedTime } from 'date-fns-tz'
import { addDays, isWeekend, getHours, format } from 'date-fns'

const SA_TZ = 'Africa/Johannesburg'
const CUTOFF_HOUR = 17 // 17:00 SAST

/**
 * Returns the business day on which the onboarding timer should start.
 * Rule: signups Mon-Thu before 17:00 SAST → today.
 *       Fri or weekday after 17:00 SAST → next Monday.
 *       Sat/Sun → next Monday.
 * Result is a YYYY-MM-DD date (no time component) in SA timezone.
 */
export function computeTimerStartDay(signupAt: Date = new Date()): string {
  let saTime = toZonedTime(signupAt, SA_TZ)
  
  // If after cutoff hour, push to next day
  if (getHours(saTime) >= CUTOFF_HOUR) {
    saTime = addDays(saTime, 1)
  }
  // Skip weekends
  while (isWeekend(saTime)) {
    saTime = addDays(saTime, 1)
  }
  return format(saTime, 'yyyy-MM-dd')
}

/**
 * Adds N business days (Mon-Fri only, no public holidays in v3.0) to a YYYY-MM-DD date.
 */
export function addBusinessDays(startDay: string, businessDays: number): string {
  let d = new Date(startDay + 'T00:00:00')
  let added = 0
  while (added < businessDays) {
    d = addDays(d, 1)
    if (!isWeekend(d)) added++
  }
  return format(d, 'yyyy-MM-dd')
}
```

```typescript
import { getAdminClient } from '@/lib/supabase/admin'

export type SagaStatus = 'pending'|'running'|'completed'|'failed'|'rolled_back'|'paused'

export interface SagaState {
  jobId: string
  organizationId: string
  status: SagaStatus
  currentStep: number
  stepsCompleted: string[]
  createdResources: Record<string, unknown>
  errorMessage: string | null
}

export async function pauseSaga(jobId: string, currentStep: number, errorMessage: string, createdResources: Record<string, unknown>): Promise<void> {
  const supa = getAdminClient()
  await supa.from('provisioning_jobs').update({
    status: 'paused',
    current_step: currentStep,
    error_message: errorMessage,
    created_resources: createdResources,
    updated_at: new Date().toISOString(),
  }).eq('id', jobId)

  // Telegram alert — non-blocking, swallow failures (we still want the saga to pause)
  await sendOpsTelegramAlert({
    title: 'Provisioning saga PAUSED',
    body: `Job ${jobId} paused at step ${currentStep}.\nError: ${errorMessage}\nResume via /api/ops/provisioning-resume {"jobId":"${jobId}"}`,
  }).catch((err) => console.error('Telegram alert failed:', err))
}

export async function resumeSaga(jobId: string): Promise<SagaState> {
  const supa = getAdminClient()
  const { data, error } = await supa.from('provisioning_jobs').select('*').eq('id', jobId).single()
  if (error || !data) throw new Error(`Job ${jobId} not found`)
  if (data.status !== 'paused') throw new Error(`Job ${jobId} is ${data.status}, not paused — cannot resume`)
  await supa.from('provisioning_jobs').update({ status: 'running' }).eq('id', jobId)
  return {
    jobId: data.id,
    organizationId: data.organization_id,
    status: 'running',
    currentStep: data.current_step,
    stepsCompleted: data.steps_completed ?? [],
    createdResources: data.created_resources ?? {},
    errorMessage: null,
  }
}

async function sendOpsTelegramAlert(opts: { title: string; body: string }): Promise<void> {
  const token = process.env.TELEGRAM_BOT_TOKEN
  const chatId = process.env.TELEGRAM_OPS_CHAT_ID
  if (!token || !chatId) return // env not set in dev — silent
  await fetch(`https://api.telegram.org/bot${token}/sendMessage`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ chat_id: chatId, text: `${opts.title}\n\n${opts.body}`, parse_mode: 'Markdown' }),
  })
}
```

**1.6: `app/api/ops/provisioning-resume/route.ts`**

```typescript
import { NextRequest } from 'next/server'
import { getUserOrg } from '@/lib/auth/get-user-org'
import { resumeSaga } from '@/lib/provisioning/saga-state'

export async function POST(req: NextRequest) {
  const { data: userOrg, error: authError } = await getUserOrg()
  if (authError || !userOrg) return Response.json({ error: 'unauthenticated' }, { status: 401 })
  // Platform admin only
  if (userOrg.role !== 'platform_admin' && userOrg.role !== 'admin') {
    return Response.json({ error: 'forbidden' }, { status: 403 })
  }
  const { jobId } = await req.json()
  if (!jobId) return Response.json({ error: 'jobId_required' }, { status: 400 })

  try {
    const state = await resumeSaga(jobId)
    // The current orchestrator (`scripts/provisioning/orchestrator.ts`) exports a single
    // linear `provisionClient(...)` function — it is NOT pausable mid-flight, and
    // re-running steps 5-9 over HTTP can exceed Vercel's 60s function limit (especially
    // step 5: N8N workflow create per module).
    //
    // Therefore: this endpoint flips status to 'running' and returns 202 with a CLI
    // continuation hint. The operator runs `pnpm run provisioning:resume <jobId>` from
    // their dev machine (which has unbounded runtime) to drive the actual step replay.
    // Steps 5-9 are designed idempotent (Pattern 3 in RESEARCH.md) so re-running from
    // current_step is safe.
    return Response.json({
      ok: true,
      status: 'resumed',
      jobId: state.jobId,
      currentStep: state.currentStep,
      next_action: `Run: pnpm run provisioning:resume ${state.jobId}`,
    }, { status: 202 })
  } catch (err) {
    return Response.json({ error: 'resume_failed', detail: (err as Error).message }, { status: 500 })
  }
}
```

**1.6b: `scripts/provisioning/resume.ts` — CLI helper that continues the saga**

The HTTP route above flips status to 'running' but returns immediately. This CLI does the actual replay of steps from `current_step` onward. It runs locally (no HTTP timeout) and uses the SAME step functions as orchestrator.ts.

```typescript
#!/usr/bin/env node
/**
 * Resume a paused provisioning saga from its current_step.
 * Usage: pnpm run provisioning:resume <jobId>
 *
 * Reads provisioning_jobs row for the jobId, validates status='running' (set by
 * /api/ops/provisioning-resume), then invokes step functions from current_step
 * onward. Each step is idempotent — re-running a partially-completed step is safe.
 */
import { getAdminClient } from '../../lib/supabase/admin'
// Import the step functions we need to replay (same set the orchestrator uses)
import { createN8NWorkflow } from './steps/05-n8n'
import { deployAutomations } from './steps/06-automations'
import { sendOnboardingSequence } from './steps/07-onboarding'
import { runQAChecks } from './steps/08-qa-check'
import { pauseSaga } from '../../lib/provisioning/saga-state'

async function main() {
  const jobId = process.argv[2]
  if (!jobId) { console.error('Usage: pnpm run provisioning:resume <jobId>'); process.exit(1) }

  const supa = getAdminClient()
  const { data: job, error } = await supa.from('provisioning_jobs').select('*').eq('id', jobId).single()
  if (error || !job) { console.error('Job not found:', error?.message); process.exit(1) }
  if (job.status !== 'running') { console.error(`Job is ${job.status}, not running. Call /api/ops/provisioning-resume first.`); process.exit(1) }

  const resources = (job.created_resources as Record<string, unknown>) ?? {}
  const stepsArray = [
    { num: 5, name: 'n8n', fn: createN8NWorkflow },
    { num: 6, name: 'automations', fn: deployAutomations },
    { num: 7, name: 'onboarding', fn: sendOnboardingSequence },
    { num: 8, name: 'qa', fn: runQAChecks },
    // Step 10 (schedule-followups) added in 1.4 — append it here once orchestrator is wired
  ]

  for (const step of stepsArray.filter((s) => s.num >= job.current_step)) {
    console.log(`Resuming step ${step.num} (${step.name})...`)
    try {
      // Reconstruct the minimal `job` object the step functions expect — read orchestrator.ts to confirm shape
      const stepJob = { clientId: job.client_id, clientName: job.client_name, orgEmail: job.org_email, tier: job.tier, clientConfig: job.client_config, createdResources: resources }
      const result = await step.fn(stepJob as any, resources as any)
      if (!result || (result as any).success === false) {
        throw new Error(`Step ${step.num} failed: ${(result as any)?.error ?? 'unknown'}`)
      }
      Object.assign(resources, (result as any).data ?? {})
      await supa.from('provisioning_jobs').update({ current_step: step.num + 1, created_resources: resources }).eq('id', jobId)
    } catch (err) {
      console.error(`Step ${step.num} threw:`, err)
      await pauseSaga(jobId, step.num, (err as Error).message, resources)
      process.exit(1)
    }
  }

  await supa.from('provisioning_jobs').update({ status: 'completed' }).eq('id', jobId)
  console.log('Resume complete. Job marked completed.')
}

main().catch((e) => { console.error(e); process.exit(1) })
```

**2.2: `lib/onboarding/progress.ts`**

```typescript
import { getAdminClient } from '@/lib/supabase/admin'

export async function recordEmailSent(orgId: string, day: 0|1|2|3): Promise<void> {
  const supa = getAdminClient()
  const col = day === 0 ? 'day0_completed_at' : `day${day}_email_sent_at`
  await supa.from('onboarding_progress').update({
    [col]: new Date().toISOString(),
  }).eq('organization_id', orgId)
}

export async function recordDriftFlag(orgId: string, flag: string): Promise<void> {
  // No RPC call — array_append_drift_flag does NOT exist in the DB. Use read-append-update.
  // Drift flags are advisory (operator notification, not transactional state); the small
  // race window between SELECT and UPDATE is acceptable. If a concurrent writer beats us,
  // we lose at most one flag entry — not data integrity.
  const supa = getAdminClient()
  const { data: existing } = await supa
    .from('onboarding_progress')
    .select('drift_flags')
    .eq('organization_id', orgId)
    .maybeSingle()
  const current: string[] = (existing?.drift_flags as string[] | null) ?? []
  if (current.includes(flag)) return // dedupe — don't append if already present
  const next = [...current, flag]
  await supa
    .from('onboarding_progress')
    .update({ drift_flags: next })
    .eq('organization_id', orgId)
}

export async function getOrgOnboardingProgress(orgId: string) {
  const supa = getAdminClient()
  const { data } = await supa.from('onboarding_progress').select('*').eq('organization_id', orgId).maybeSingle()
  return data
}
```

**2.3: `app/api/n8n/onboarding-day/route.ts` — N8N WEBHOOK TARGET**

This endpoint is hit by N8N Day 1/2/3 cron workflows. Authenticated via shared secret (NOT user auth — N8N is a service):

```typescript
import { NextRequest } from 'next/server'
import { sendOnboardingDay1 } from '@/emails/onboarding-day1'
import { sendOnboardingDay2 } from '@/emails/onboarding-day2'
import { sendOnboardingDay3 } from '@/emails/onboarding-day3'
import { recordEmailSent, recordDriftFlag } from '@/lib/onboarding/progress'

export async function POST(req: NextRequest) {
  // Constant-time shared-secret check
  const provided = req.headers.get('x-n8n-secret')
  const expected = process.env.N8N_WEBHOOK_SECRET
  if (!expected || !provided || provided !== expected) {
    return Response.json({ error: 'unauthorized' }, { status: 401 })
  }

  const { orgId, day } = await req.json()
  if (!orgId || ![1,2,3].includes(day)) return Response.json({ error: 'invalid_input' }, { status: 400 })

  let err: string | null = null
  if (day === 1) err = await sendOnboardingDay1(orgId)
  else if (day === 2) err = await sendOnboardingDay2(orgId)
  else if (day === 3) err = await sendOnboardingDay3(orgId)

  if (err) {
    await recordDriftFlag(orgId, `day${day}_email_failed`)
    return Response.json({ error: 'send_failed', detail: err }, { status: 500 })
  }
  await recordEmailSent(orgId, day)
  return Response.json({ ok: true, day })
}
```

**2.4: 3 N8N workflow JSONs `n8n/onboarding-day1.json`, `n8n/onboarding-day2.json`, `n8n/onboarding-day3.json`**

Each is a cron workflow that:
1. Triggers daily at 10:00 SAST (`0 8 * * *` UTC = 10:00 SAST in winter; document this)
2. Queries Supabase for orgs at the relevant day:
   - Day 1: `WHERE timer_start_day = CURRENT_DATE AND day1_email_sent_at IS NULL`
   - Day 2: `WHERE timer_start_day = (CURRENT_DATE - 1 business_day) AND day2_email_sent_at IS NULL`
   - Day 3: `WHERE timer_start_day = (CURRENT_DATE - 2 business_day) AND day3_email_sent_at IS NULL`
3. For each org, POSTs to `https://draggonnb-platform.vercel.app/api/n8n/onboarding-day` with `{orgId, day}` and `x-n8n-secret` header

Use the existing `n8n/` workflow JSON structure as a template (read existing files for shape — SUPABASE_QUERY → SPLIT_IN_BATCHES → HTTP_REQUEST). Day-2 / Day-3 weekend skipping is achieved by computing the target date in JavaScript code node:
```javascript
const today = new Date()
let businessDaysBack = N // 1, 2, or 3
let target = new Date(today)
while (businessDaysBack > 0) {
  target.setDate(target.getDate() - 1)
  const dow = target.getDay()
  if (dow !== 0 && dow !== 6) businessDaysBack--
}
return [{ json: { target_day: target.toISOString().split('T')[0] } }]
```
