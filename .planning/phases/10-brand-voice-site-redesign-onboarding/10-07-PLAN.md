```typescript
import { formatInTimeZone } from 'date-fns-tz'

const SA_TZ = 'Africa/Johannesburg'

/**
 * Formats a UTC ISO timestamp as the user-facing reset display.
 * USAGE-04 spec: "exact reset date/time in tenant timezone".
 * v3.0 hardcodes Africa/Johannesburg (single-country deployment); future i18n
 * can swap on tenant locale field.
 *
 * Format: 'd MMMM at HH:mm SAST'  e.g.  '1 May at 00:00 SAST'
 */
export function formatResetTimestamp(date: Date | string): string {
  const d = typeof date === 'string' ? new Date(date) : date
  return formatInTimeZone(d, SA_TZ, "d MMMM 'at' HH:mm 'SAST'")
}

/**
 * Returns the start-of-next-month timestamp in Africa/Johannesburg as ISO UTC.
 * Used as the default reset target for monthly usage caps.
 */
export function nextMonthlyResetUTC(now: Date = new Date()): Date {
  // Compute the SA-local first-of-next-month at 00:00:00, then convert to UTC
  const saNow = formatInTimeZone(now, SA_TZ, 'yyyy-MM-dd')
  const [y, m] = saNow.split('-').map(Number)
  const nextMonth = m === 12 ? `${y + 1}-01-01T00:00:00+02:00` : `${y}-${String(m + 1).padStart(2, '0')}-01T00:00:00+02:00`
  return new Date(nextMonth)
}
```

**1b.1: `app/api/usage/current/route.ts` — lightweight GET endpoint**

Returns the calling org's currently-metered metrics with usage percentages. Read from `usage_events` (current period sum) joined against `billing_plans.limits` (the org's plan caps). Phase 09's USAGE-13 cleanup means `client_usage_metrics` is gone, so we query the source of truth (`usage_events`) directly.

```typescript
import { getUserOrg } from '@/lib/auth/get-user-org'
import { getAdminClient } from '@/lib/supabase/admin'

interface MetricSnapshot {
  metric: string
  current: number
  limit: number
  percent: number
}

export async function GET() {
  const { data: userOrg, error: authError } = await getUserOrg()
  if (authError || !userOrg) return Response.json({ error: 'unauthenticated' }, { status: 401 })

  const supa = getAdminClient()

  // 1. Fetch the org's plan limits JSONB (e.g. { agent_invocations: 1000, email_sends: 5000, ... })
  const { data: org } = await supa
    .from('organizations')
    .select('subscription_tier')
    .eq('id', userOrg.organizationId)
    .single()
  if (!org) return Response.json([], { status: 200 })

  const { data: plan } = await supa
    .from('billing_plans')
    .select('limits')
    .eq('tier', org.subscription_tier)
    .maybeSingle()
  const limits: Record<string, number> = (plan?.limits as Record<string, number>) ?? {}

  // 2. Compute period start (1st of current month, UTC — close enough for v3.0; future: align to SAST)
  const now = new Date()
  const periodStart = new Date(Date.UTC(now.getUTCFullYear(), now.getUTCMonth(), 1, 0, 0, 0))

  // 3. For each metric in limits, sum usage_events for current period
  const metrics: MetricSnapshot[] = []
  for (const [metric, limit] of Object.entries(limits)) {
    if (typeof limit !== 'number' || limit <= 0) continue
    const { data: rows } = await supa
      .from('usage_events')
      .select('quantity')
      .eq('organization_id', userOrg.organizationId)
      .eq('metric', metric)
      .gte('created_at', periodStart.toISOString())
    const current = (rows ?? []).reduce((sum, r) => sum + (r.quantity ?? 0), 0)
    const percent = Math.round((current / limit) * 100)
    metrics.push({ metric, current, limit, percent })
  }

  return Response.json(metrics)
}
```

Implementation notes:
- The metric names must match what guardUsage writes — verify against `lib/usage/guard.ts` constants. If billing_plans.limits keys diverge from guardUsage's metric names, this endpoint silently returns nothing — log a warning and surface in SUMMARY if mismatch found.
- This endpoint is RSC-fetched (server-side), not client-fetched — keeps the banner server-rendered and avoids a client-side waterfall.
- For v3.0, period boundaries hardcode UTC start-of-month. A future improvement converts to Africa/Johannesburg (matching USAGE-04 reset display).

```typescript
import { UsageWarningBanner, thresholdFor } from './_components/usage-warning-banner'
import { headers } from 'next/headers'

interface MetricSnapshot { metric: string; current: number; limit: number; percent: number }

async function fetchCurrentUsage(): Promise<MetricSnapshot[]> {
  // Build absolute URL for server-side fetch
  const h = await headers()
  const host = h.get('host') ?? 'draggonnb-platform.vercel.app'
  const proto = host.includes('localhost') ? 'http' : 'https'
  // Forward cookie for auth (RSC fetch needs the user's session)
  const cookie = h.get('cookie') ?? ''
  const res = await fetch(`${proto}://${host}/api/usage/current`, {
    headers: { cookie },
    cache: 'no-store',
  })
  if (!res.ok) return []
  return res.json()
}

export default async function DashboardLayout({ children }: { children: React.ReactNode }) {
  const usage = await fetchCurrentUsage()
  const banners = usage
    .map((m) => ({ metric: m.metric, used: m.current, limit: m.limit, threshold: thresholdFor(m.current, m.limit) }))
    .filter((b): b is { metric: string; used: number; limit: number; threshold: 0.50|0.75|0.90 } => b.threshold !== null)

        <div className="px-6 pt-4 space-y-2">
          {banners.map((b) => (
            <UsageWarningBanner key={b.metric} {...b} />
          ))}
        </div>
      )}
      {/* existing children render */}
      {children}
    </div>
  )
}
```

**2.3: `app/api/admin/orgs/[id]/archive/route.ts` — admin UI archive endpoint**

```typescript
import { NextRequest } from 'next/server'
import { getUserOrg } from '@/lib/auth/get-user-org'
import { getAdminClient } from '@/lib/supabase/admin'

const PRESERVE_ORG_NAME = 'DragoonB Business Automation'

export async function POST(req: NextRequest, { params }: { params: Promise<{ id: string }> }) {
  const { data: userOrg, error } = await getUserOrg()
  if (error || !userOrg) return Response.json({ error: 'unauthenticated' }, { status: 401 })
  if (userOrg.role !== 'admin' && userOrg.role !== 'platform_admin') return Response.json({ error: 'forbidden' }, { status: 403 })

  const { id } = await params
  const supa = getAdminClient()

  // Hard guard: never archive the platform_admin org
  const { data: target } = await supa.from('organizations').select('id, name').eq('id', id).single()
  if (!target) return Response.json({ error: 'not_found' }, { status: 404 })
  if (target.name === PRESERVE_ORG_NAME) {
    return Response.json({ error: 'cannot_archive_platform_admin' }, { status: 400 })
  }

  const { error: updErr } = await supa
    .from('organizations')
    .update({ archived_at: new Date().toISOString() })
    .eq('id', id)
  if (updErr) return Response.json({ error: 'archive_failed', detail: updErr.message }, { status: 500 })

  return Response.json({ ok: true })
}
```
