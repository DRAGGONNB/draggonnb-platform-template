```ts
import { createAdminClient } from '@/lib/supabase/admin'
import { MeteredMetric, UsageCapExceededError } from './types'

export interface GuardUsageOptions {
  orgId: string
  metric: MeteredMetric
  qty?: number  // default 1
  idempotencyKey?: string
}

export async function guardUsage(opts: GuardUsageOptions): Promise<void> {
  const supabase = createAdminClient()
  const { data, error } = await supabase.rpc('record_usage_event', {
    p_organization_id: opts.orgId,
    p_metric: opts.metric,
    p_qty: opts.qty ?? 1,
    p_idempotency_key: opts.idempotencyKey ?? null,
  })

  if (error) {
    // RPC exec error — propagate (caller will treat as 5xx)
    throw error
  }

  if (data === false) {
    // Cap exceeded — read current usage for error context
    // Single follow-up query is acceptable; only on the failure path
    const { data: usage } = await supabase
      .from('usage_events')
      .select('qty')
      .eq('organization_id', opts.orgId)
      .eq('metric', opts.metric)
      .gte('created_at', firstOfMonth())

    const total = (usage ?? []).reduce((s, r) => s + (r.qty ?? 0), 0)
    const limit = await fetchTierLimit(opts.orgId, opts.metric)
    throw new UsageCapExceededError(opts.orgId, opts.metric, total, limit)
  }
}

function firstOfMonth(): string {
  const now = new Date()
  return new Date(now.getFullYear(), now.getMonth(), 1).toISOString()
}

async function fetchTierLimit(orgId: string, metric: MeteredMetric): Promise<number> {
  const supabase = createAdminClient()
  const { data } = await supabase
    .from('organizations')
    .select('billing_plans!inner(limits)')
    .eq('id', orgId)
    .single()
  const limits = (data as any)?.billing_plans?.limits ?? {}
  return limits[metric] ?? Infinity
}
```

**Create lib/ai/cost-calculator.ts** — pure function for ZAR cents from Anthropic usage payload.

```ts
import { ModelId, MODEL_PRICING } from './model-registry'

export interface AnthropicUsage {
  input_tokens: number
  output_tokens: number
  cache_read_input_tokens?: number      // SDK v0.73+ field
  cache_creation_input_tokens?: number  // SDK v0.73+ field
}

const USD_TO_ZAR_CENTS = 16.6 * 100  // 1660 cents per USD; configurable via env later

export function computeCostZarCents(usage: AnthropicUsage, model: ModelId): number {
  const pricing = MODEL_PRICING[model]
  if (!pricing) throw new Error(`Unknown model for pricing: ${model}`)

  const cacheReads = usage.cache_read_input_tokens ?? 0
  const cacheCreates = usage.cache_creation_input_tokens ?? 0
  const regularInputs = Math.max(0, usage.input_tokens - cacheReads - cacheCreates)

  const inputCostUsd = (regularInputs / 1_000_000) * pricing.inputCostPerMTokUsd
  const cachedReadCostUsd = (cacheReads / 1_000_000) * pricing.cachedInputCostPerMTokUsd
  const cacheCreateCostUsd = (cacheCreates / 1_000_000) * (pricing.inputCostPerMTokUsd * 1.25)
  const outputCostUsd = (usage.output_tokens / 1_000_000) * pricing.outputCostPerMTokUsd

  const totalUsd = inputCostUsd + cachedReadCostUsd + cacheCreateCostUsd + outputCostUsd
  return Math.ceil(totalUsd * USD_TO_ZAR_CENTS)
}
```

Notes:
- `Math.ceil` rounds up to whole cents — protects margin
- Cache write cost = 1.25× input cost (Anthropic spec)
- USD/ZAR rate is hardcoded at 16.6; OPS-01 (env validation) will surface a future ANTHROPIC_USD_ZAR_RATE env var. Acceptable hard-code for v3.0.

```ts
import { createAdminClient } from '@/lib/supabase/admin'

export const TIER_CEILING_ZAR_CENTS: Record<string, number> = {
  core: 15_000,        // R150
  growth: 40_000,      // R400
  scale: 150_000,      // R1,500
  platform_admin: Number.MAX_SAFE_INTEGER,  // unlimited for Chris's own org
}

export class CostCeilingExceededError extends Error {
  constructor(
    public readonly orgId: string,
    public readonly mtdSpendCents: number,
    public readonly projectedCents: number,
    public readonly ceilingCents: number,
  ) {
    super(`Cost ceiling exceeded for org ${orgId}: ${mtdSpendCents + projectedCents}/${ceilingCents} ZAR cents`)
    this.name = 'CostCeilingExceededError'
  }
}

export async function checkCostCeiling(
  orgId: string,
  projectedCostZarCents: number,
): Promise<void> {
  const supabase = createAdminClient()

  // CANONICAL TIER RESOLUTION:
  // - Read organizations.plan_id (FK to billing_plans.id) — this is the canonical tier identity
  // - DO NOT read organizations.subscription_tier — that column contains legacy aliases (starter/professional/enterprise)
  //   that won't match TIER_CEILING_ZAR_CENTS keys and will silently fall back to 'core'
  // - Pass result through getCanonicalTierName() to normalize legacy aliases → canonical (core/growth/scale)
  const [{ data: org }, { data: mtd }] = await Promise.all([
    supabase.from('organizations').select('plan_id').eq('id', orgId).single(),
    supabase.rpc('get_month_to_date_ai_cost', { p_org_id: orgId }),
  ])

  const { getCanonicalTierName } = await import('@/lib/payments/payfast')
  const canonicalTier = getCanonicalTierName(org?.plan_id ?? 'core')
  const ceiling = TIER_CEILING_ZAR_CENTS[canonicalTier] ?? TIER_CEILING_ZAR_CENTS.core
  const mtdSpend = (mtd as number) ?? 0

  if (mtdSpend + projectedCostZarCents > ceiling) {
    throw new CostCeilingExceededError(orgId, mtdSpend, projectedCostZarCents, ceiling)
  }
}

/**
 * Estimate projected cost for a request based on input length only.
 * Output tokens unknown until response, so use conservative max_tokens estimate.
 */
export function projectCost(
  inputTokensEstimate: number,
  maxOutputTokens: number,
  model: import('./model-registry').ModelId,
): number {
  // Use cost calculator with worst-case usage (no cache hits projected)
  const { computeCostZarCents } = require('./cost-calculator')
  return computeCostZarCents(
    { input_tokens: inputTokensEstimate, output_tokens: maxOutputTokens },
    model,
  )
}
```

Notes:
- `get_month_to_date_ai_cost` RPC is created in 09-01 (migration 27) — verify it exists before this plan executes
- Tier resolution: ALWAYS read `organizations.plan_id` and pass through `getCanonicalTierName()`. Migration 11 created `billing_plans` with canonical IDs (`core`, `growth`, `scale`). Do NOT read `subscription_tier` (legacy column with old `starter`/`professional`/`enterprise` aliases that hash-miss in `TIER_CEILING_ZAR_CENTS`).
- `getCanonicalTierName` already exists in `lib/payments/payfast.ts` — reuse it.
