```sql
-- scripts/migrations/phase-09/audit-client-usage-metrics.sql
-- Phase 09 diagnostic — Task 09-01-07
-- Read-only. Run via Supabase SQL editor or psql.
-- Produces a textual report on client_usage_metrics column-mismatch bug surface.

\echo '=== client_usage_metrics column audit ==='

-- 1. Confirm table exists and list actual columns
SELECT column_name, data_type, column_default
  FROM information_schema.columns
 WHERE table_schema = 'public' AND table_name = 'client_usage_metrics'
 ORDER BY ordinal_position;

-- 2. Row counts
SELECT COUNT(*) AS total_rows FROM client_usage_metrics;

-- 3. Rows with any non-zero counter (proves writes have landed SOMEWHERE)
SELECT COUNT(*) AS rows_with_posts, SUM(posts_monthly) AS total_posts,
       COUNT(*) FILTER (WHERE posts_monthly > 0) AS rows_posts_nonzero,
       COUNT(*) FILTER (WHERE ai_generations_monthly > 0) AS rows_ai_nonzero,
       MAX(reset_date) AS most_recent_reset,
       MIN(reset_date) AS earliest_reset
  FROM client_usage_metrics;

-- 4. Recent activity check — if webhook reset code is broken, reset_date will be stale
SELECT COUNT(*) AS rows_reset_last_7d
  FROM client_usage_metrics
 WHERE reset_date > now() - interval '7 days';

-- 5. Cross-check with usage_events — do we have parallel writes?
SELECT COUNT(DISTINCT organization_id) AS orgs_in_usage_events,
       (SELECT COUNT(DISTINCT organization_id) FROM client_usage_metrics) AS orgs_in_legacy_metrics
  FROM usage_events
 WHERE created_at > now() - interval '30 days';

-- 6. Orgs writing to BOTH systems (dual-state confirmation)
SELECT o.id, o.name, o.subscription_tier,
       c.posts_monthly AS legacy_posts, c.ai_generations_monthly AS legacy_ai,
       (SELECT COUNT(*) FROM usage_events ue WHERE ue.organization_id = o.id) AS new_events_count
  FROM organizations o
  LEFT JOIN client_usage_metrics c ON c.organization_id = o.id
 ORDER BY o.created_at DESC
 LIMIT 20;

\echo '=== audit complete — capture output for 09-DIAGNOSTICS.md in Plan 09-05 ==='
```
