```sql
-- Phase 13 Plan 07 — RPC for safe JSONB path-set merging.
-- Used by activate-trophy-module saga step to update tenant_modules.config.trophy.linked_org_id
-- without replacing the entire JSONB blob.

CREATE OR REPLACE FUNCTION set_tenant_module_config_path(
  p_organization_id UUID,
  p_module_id TEXT,
  p_path TEXT[],
  p_value TEXT
)
RETURNS VOID
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
BEGIN
  UPDATE tenant_modules
  SET config = jsonb_set(
    COALESCE(config, '{}'::jsonb),
    p_path,
    to_jsonb(p_value),
    true  -- create_missing
  )
  WHERE organization_id = p_organization_id
    AND module_id = p_module_id;

  IF NOT FOUND THEN
    RAISE EXCEPTION 'No tenant_modules row for org=% module=%', p_organization_id, p_module_id;
  END IF;
END;
$$;

REVOKE EXECUTE ON FUNCTION set_tenant_module_config_path FROM PUBLIC;
REVOKE EXECUTE ON FUNCTION set_tenant_module_config_path FROM anon;
REVOKE EXECUTE ON FUNCTION set_tenant_module_config_path FROM authenticated;
```

```typescript
// scripts/provisioning/steps/activate-trophy-module.ts
// SSO-11: Provisioning saga step 10 — activate Trophy module for a DraggonnB org.
// Idempotent, retryable, paused-on-failure (existing saga semantics).
// D6: auto-create Trophy `orgs` row at module-activation time, NOT at first SSO bridge.

import { createAdminClient } from '@/lib/supabase/admin'
import type { ProvisioningJob, ProvisioningResult } from '@/lib/provisioning/types'

export async function activateTrophyModule(job: ProvisioningJob): Promise<ProvisioningResult> {
  const admin = createAdminClient()
  const draggonnbOrgId = job.createdResources?.organizationId

  if (!draggonnbOrgId) {
    return { success: false, step: 'activate-trophy-module', error: 'No organizationId in job.createdResources' }
  }

  // Idempotency: if already linked, no-op
  const { data: existing } = await admin
    .from('organizations')
    .select('linked_trophy_org_id')
    .eq('id', draggonnbOrgId)
    .single()

  if (existing?.linked_trophy_org_id) {
    return {
      success: true,
      step: 'activate-trophy-module',
      data: { trophyOrgId: existing.linked_trophy_org_id },
    }
  }

  // 1. Create Trophy `orgs` row.
  // CONFIRMED enum value for `type`: '<value chosen in step 1>' — see Phase 13 plan 07 hand-off.
  const trophyType = '<chosen-enum-value>'  // <-- replace with actual chosen value
  const { data: trophyOrg, error: trophyOrgError } = await admin
    .from('orgs')
    .insert({
      name: job.clientName,
      slug: job.createdResources?.subdomain || job.clientId,
      type: trophyType,
      subscription_status: 'trial',
    })
    .select('id')
    .single()

  if (trophyOrgError || !trophyOrg) {
    return {
      success: false,
      step: 'activate-trophy-module',
      error: `Trophy orgs row insert failed: ${trophyOrgError?.message ?? 'unknown'}`,
    }
  }

  // 2. Cross-product link row.
  const { error: linkError } = await admin
    .from('cross_product_org_links')
    .insert({
      draggonnb_org_id: draggonnbOrgId,
      trophy_org_id: trophyOrg.id,
      status: 'active',
    })

  if (linkError) {
    // Rollback the Trophy orgs row to keep state consistent
    await admin.from('orgs').delete().eq('id', trophyOrg.id)
    return {
      success: false,
      step: 'activate-trophy-module',
      error: `cross_product_org_links insert failed: ${linkError.message}`,
    }
  }

  // 3. Update organizations.linked_trophy_org_id (FK enforced as of plan 13-07 step 4 migration).
  const { error: updateError } = await admin
    .from('organizations')
    .update({ linked_trophy_org_id: trophyOrg.id })
    .eq('id', draggonnbOrgId)

  if (updateError) {
    // Rollback both
    await admin.from('cross_product_org_links').delete().eq('trophy_org_id', trophyOrg.id)
    await admin.from('orgs').delete().eq('id', trophyOrg.id)
    return {
      success: false,
      step: 'activate-trophy-module',
      error: `organizations update failed: ${updateError.message}`,
    }
  }

  // 4. JSONB cache: tenant_modules.config.trophy.linked_org_id
  const { error: rpcError } = await admin.rpc('set_tenant_module_config_path', {
    p_organization_id: draggonnbOrgId,
    p_module_id: 'trophy',
    p_path: ['trophy', 'linked_org_id'],
    p_value: trophyOrg.id,
  })

  if (rpcError) {
    // Best-effort: log but do not roll back — the FK + link table are the source of truth.
    // The JSONB is a denormalized cache for sidebar conditional rendering performance.
    // Sidebar will fall back to organizations.linked_trophy_org_id if config is empty.
    console.error('[activate-trophy-module] JSONB cache update failed (non-fatal):', rpcError.message)
  }

  return {
    success: true,
    step: 'activate-trophy-module',
    data: { trophyOrgId: trophyOrg.id },
  }
}
```

WHY rollback explicitly on each step instead of relying on saga rollback: The saga rollback fires on the OUTER step's failure return. Within the step, partial state must be cleaned up so a retry sees a consistent baseline. The idempotency check at the top handles the happy retry case.

WHY JSONB write is best-effort but FK + link are critical: The sidebar reads the FK column (fast, indexed) as the source of truth; the JSONB is a denormalization for legacy NAV checks. If the JSONB fails, the conditional rendering still works.

```typescript
// app/api/activate-trophy/route.ts
// SSO-11: explicit user-triggered activation. Calls the saga step directly.

import { NextResponse } from 'next/server'
import { getUserOrg } from '@/lib/auth/get-user-org'
import { activateTrophyModule } from '@/scripts/provisioning/steps/activate-trophy-module'

export async function POST() {
  const { data: userOrg, error } = await getUserOrg()
  if (error || !userOrg) {
    return NextResponse.json({ error: 'Authentication required' }, { status: 401 })
  }

  // Only org admins can activate
  if (userOrg.role !== 'admin') {
    return NextResponse.json({ error: 'Only org admins can activate Trophy OS' }, { status: 403 })
  }

  const result = await activateTrophyModule({
    clientId: userOrg.organizationId,
    clientName: userOrg.organizationName ?? 'Untitled',
    createdResources: {
      organizationId: userOrg.organizationId,
      subdomain: userOrg.subdomain ?? null,
    },
  } as any)  // ProvisioningJob shape may need adjustment — see hand-off note below

  if (!result.success) {
    return NextResponse.json({ error: result.error }, { status: 500 })
  }

  return NextResponse.json({ ok: true, trophyOrgId: result.data?.trophyOrgId })
}
```
